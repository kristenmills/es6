var vs let

Worse People

              
function fooBar() {
  var bar = 2;
  if(true) {
    var bar = 1;
  }
  return bar;
}

Better People

              
function fooBar() {
  const bar = 2;
  if(true) {
    let bar = 1;
  }
  return bar;
}

Arrow Functions

Worse People

              
function FooBar(){
  this.multiplier = 3;
  this.data = [1,2,3];
  var other = this;
  this.data = [1,2,3].map(function(item){
    return other.multiplier * item;
  });
  return this.data;
}
new FooBar();

Better People

              
function FooBar(){
  this.multiplier = 3;
  this.data = [1,2,3];
  .map((item) => {
    return this.multiplier * item;
  });
  return this.data;
}
new FooBar();

Classes

"Worse" People

              
function FooBar() {
  this.foo = 1;
  this.bar = 2;
}

FooBar.prototype.add = function() {
  return this.foo + this.bar;
}

var myFooBar = Object.create(FooBar.prototype);
myFooBar.add();

Better People

              
class FooBar {
  constructor(){
    this.foo = 1;
    this.bar = 2;
  }

  add() {
    return this.foo + this.bar;
  }
}
const myFooBar = new FooBar();
myFooBar.add();

Rest and Spread

Worse People

              
function numArgs() {
  return arguments.length;
}

function sum(x, y, z) {
  return x + y + z;
}

numArgs(1,2,3)
var array = [1,2,3];
sum(array[0], array[1], array[2])

Better People

              
function numArgs(...y) {
  return y.length;
}

function sum(x, y, z) {
  return x + y + z;
}

numArgs(1,2,3);
const array = [1,2,3];
sum(...array);

Iterators and for...of

            
let fibonacci = {
  [Symbol.iterator]() {
    let pre = 0, cur = 1;
    return {
      next() {
        [pre, cur] = [cur, pre + cur];
        return { done: false, value: cur }
      }
    }
  }
}

for (var n of fibonacci) {
  // truncate the sequence at 1000
  if (n > 1000)
    break;
  console.log(n);
}

Generators

            
let fibonacci = {
  [Symbol.iterator]: function*() {
    let pre = 0, cur = 1;
    for (;;) {
      let temp = pre;
      pre = cur;
      cur += temp;
      yield cur;
    }
  }
}}

for (var n of fibonacci) {
  // truncate the sequence at 1000
  if (n > 1000)
    break;
  console.log(n);
}

Proxies

            
var handler = {
  get(target, name){
    return name in target ?
      target[name] :
      37;
  }
};

var p = new Proxy({}, handler);
p.a = 1
p.b = 2

console.log(p.a, p.b, p.c); // 1, 2, 37
